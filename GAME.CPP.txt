void Game::Init() {

	Solid* teapot1 = new Teapot (Vector3D(-1.0, 1.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(180.0, 180.0, 90.0), 0.2);
	Solid* teapot2 = new Teapot(Vector3D(-1.0, 0.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(180.0, 180.0, 90.0), 0.4);

	Solid* cube1 = new Cube(Vector3D(0.0, 1.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(45.0, 45.0, 45.0), 0.2);
	Solid* cube2 = new Cube(Vector3D(0.0, 0.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(45.0, 45.0, 45.0), 0.4);
	
	Solid* sphere1 = new Sphere(Vector3D(1.0, 1.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(0.0, 45.0, 0.0), 0.2, 20, 20);
	Solid* sphere2 = new Sphere(Vector3D(1.0, 0.0, -2.0), Color(1.0, 0.5, 0.6), Vector3D(0.0, 45.0, 0.0), 0.4, 20, 20);

	this->gameObjects.push_back(teapot1);
	this->gameObjects.push_back(teapot2);
	this->gameObjects.push_back(cube1);
	this->gameObjects.push_back(cube2);
	this->gameObjects.push_back(sphere1);
	this->gameObjects.push_back(sphere2);
}

void Game::Render() {

	for (int i = 0; i < gameObjects.size(); i++) {
		gameObjects[i]->Render();
	}
	
}
